-- Create a ScreenGui instance
local gui = Instance.new("ScreenGui")
gui.Name = "OpaqueGui"
gui.ResetOnSpawn = false
gui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

-- Create a Frame to cover the screen
local frame = Instance.new("Frame")
frame.Name = "OpaqueFrame"
frame.Size = UDim2.new(1, 0, 1, 0)
frame.BackgroundColor3 = Color3.new(0, 0, 0) -- Black color
frame.BackgroundTransparency = 1 -- Initially transparent
frame.Parent = gui

-- Create a TextLabel to display the message
local textLabel = Instance.new("TextLabel")
textLabel.Name = "MessageLabel"
textLabel.Size = UDim2.new(1, 0, 1, 0)
textLabel.BackgroundColor3 = Color3.new(0, 0, 0)
textLabel.BackgroundTransparency = 0.5
textLabel.Text = "Vercel V5 is down.\nReason: Vercel got copyrighted.\nFor more info, paste the link in your clipboard in a browser."
textLabel.Font = Enum.Font.SourceSans
textLabel.TextColor3 = Color3.new(1, 1, 1)
textLabel.TextSize = 20
textLabel.Parent = frame

-- Function to make the frame opaque
local function makeOpaque()
    frame.BackgroundTransparency = 0
end

-- Wait for 10 seconds
wait(10)

-- Make the frame opaque
makeOpaque()

-- Clean up the GUI after 10 seconds
gui:Destroy()
